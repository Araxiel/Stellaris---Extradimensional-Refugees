# Evacuation gate
d_evacuation_gate = {
	is_for_colonizeable = yes
	icon = "d_drifting_gateway"
	category = deposit_cat_rare

	potential = { always = no }

	planet_modifier = {
		planet_max_districts_add = -1
	}

	resources = {
		category = planet_deposits
		produces = {
			trigger = {
				planet = { has_district = district_rw_farming	}
			}
			volatile_motes = 2
			energy = 10
		}
		produces = {
			society_research = 8
		}
	}

}

d_evacuation_gate_deactivated = {
	time = 100

	icon = "d_drifting_gateway"

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = {
		always = no
	}

	planet_modifier = {
		planet_max_districts_add = -1
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 300
		}
	}

	on_cleared = {
		if = {
			limit = {
				exists = owner
				owner = {
					has_ancrel = yes
				}
			}

			custom_tooltip = minor_artifacts_found
			random_list = {
				33 = {
					owner = { add_resource = { minor_artifacts = 4 } }
				}
				33 = {
					owner = { add_resource = { minor_artifacts = 8 } }
				}
				33 = {
					owner = { add_resource = { minor_artifacts = 16 } }
				}
			}
			owner = { add_resource = { alloys = 50 } }
			else = {
				owner = { add_resource = { alloys = 100 } }
			}
		}
	}

}
